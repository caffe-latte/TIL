# 선형 자료구조 (6) - Linked List 역순 저장

1. Linked List 역순 저장
```python
class ListNode(object):
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

# 재귀 구조로 뒤집기
def reverseList_RC(head: ListNode) -> ListNode:
    def reverse(node: ListNode, prev: ListNode = None):
        if not node:
            return prev
        next, node.next = node.next, prev
        return reverse(next, node)
    return reverse(head)

head = ListNode(1)
ex_list = [ListNode(2), ListNode(3), ListNode(4), ListNode(5)]
head.next = ex_list[0]
ex_list[0].next = ex_list[1]
ex_list[1].next = ex_list[2]
ex_list[2].next = ex_list[3]
rhead = reverseList_RC(head)

while rhead:
    print(rhead.val)
    rhead = rhead.next
    
# 반복 구조로 뒤집기
def reverseList_RP(head: ListNode) -> ListNode:
    node, prev = head, None
    
    while node:
        next, node.next = node.next, prev
        prev, node = node, next
    
    return prev

head = ListNode(1)
ex_list = [ListNode(3), ListNode(5), ListNode(7), ListNode(9)]
head.next = ex_list[0]
ex_list[0].next = ex_list[1]
ex_list[1].next = ex_list[2]
ex_list[2].next = ex_list[3]
rhead = reverseList_RP(head)

while rhead:
    print(rhead.val)
    rhead = rhead.next
```
