1. Linked List Pair Swap
```python
class NodeList(object):
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next
        
# 값만 교환하는 방법
def swapPairs_V(head: ListNode) -> ListNode:
    cur = head
    while cur and cur.next:
        cur.val, cur.next.val = cur.next.val, cur.val
        cur = cur.next.next
    return head

head = ListNode(1)
list1 = ListNode(2)
list2 = ListNode(3)
list3 = ListNode(4)

head.next = list1
list1.next = list2
list2.next = list3

result = swapPairs_V(head)
while result:
    print(result.val)
    result = result.next
    
# 반복 구조를 이용하는 방법
def swapPairs_RPT(head: ListNode) -> ListNode:
    root = prev = ListNode(None)
    prev.next = head
    while head and head.next:
        b = head.next
        head.next = b.next
        b.next = head
        
        prev.next = b
        
        head = head.next
        prev = prev.next.next
        
    return root.next
    
head = ListNode(1)
list1 = ListNode(2)
list2 = ListNode(3)
list3 = ListNode(4)

head.next = list1

# 재귀
class ListNode(object):
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

def swapPairs_RC(head: ListNode) -> ListNode:
    if head and head.next:
        p = head.next
        head.next = swapPairs_RC(p.next)
        p.next = head
        return p
    return head
    
head = ListNode(1)
list1 = ListNode(2)
list2 = ListNode(3)
list3 = ListNode(4)
list4 = ListNode(5)
list5 = ListNode(6)

result = swapPairs_RC(head)
```
